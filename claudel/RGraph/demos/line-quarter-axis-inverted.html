An inverted quarter axis Line chart

This quarter axis chart is inverted so that the small area is at the top.

[No canvas support]

This goes in the documents header: 
<script src="RGraph.common.core.js"></script>
<script src="RGraph.common.effects.js"></script>
<script src="RGraph.drawing.xaxis.js"></script>
<script src="RGraph.drawing.yaxis.js"></script>
<script src="RGraph.line.js"></script>
Put this where you want the chart to show up: 
<canvas id="cvs" width="600" height="370">
    [No canvas support]
</canvas>
This is the code that generates the chart: 
<script>
    window.onload = function ()
    {
        var context = document.getElementById("cvs").getContext('2d');
        
        // Draw the background grid manually
        context.beginPath();
        context.strokeStyle = '#ddd';
            
            // Vertical grid lines
            for (var i=55; i<=576; i+=(520 / 6)) {
                context.rect(Math.round(i) + 0.5, 25, 0, 305);
            }
            
            // Horizontal grid lines
            for (var i=25; i<=345; i+=(305 / 8)) {
                context.rect(55, i + 0.5, 520, 0);
            }
        context.stroke();


        // Draw the top Y axis
        var top_yaxis = new RGraph.Drawing.YAxis({
            id: 'cvs',
            x: 55,
            options: {
                max: 10,
                numticks: 0,
                numlabels: 1,
                gutterTop: 25,
                gutterBottom: 330,
                unitsPost: 'k',
                scaleZerostart: false
            }
        }).draw()



        // Draw the bottom Y axis
        var bottom_yaxis = new RGraph.Drawing.YAxis({
            id: 'cvs',
            x: 55,
            options: {
                max: -150,
                gutterTop: 45,
                gutterBottom: 40,
                scaleInvert: true,
                numlabels: 15,
                numticks: 0,
                noendtickTop: true,
                unitsPre: '',
                unitsPost: 'k'
            }
        }).draw()

        // Draw the center X axis (which has no labels)
        var xaxis = new RGraph.Drawing.XAxis({
            id: 'cvs',
            y: 45,
            options: {
                labels: [],
                numticks: 0,
                hmargin: 25,
                gutterLeft: 55
            }
        }).draw()

        // Draw the bottom X axis (which is transparent but does have labels)
        var xaxis = new RGraph.Drawing.XAxis({
            id: 'cvs',
            y: 330,
            options: {
                labels: ['Mon','Tue','Wed','Thu','Fri','Sat','Sun'],
                labelsPosition: 'edge',
                numticks: 7,
                hmargin: 0,
                gutterLeft: 55,
                textColor: '#666',
                colors: ['rgba(0,0,0,0)'],
                xaxispos: 'center'
            }
        }).draw()


        // Draw the line chart
        var line = new RGraph.Line({
            id: 'cvs',
            data: [0,-10000,150000,0,50000,0,40000],
            options: {
                hmargin: 0,
                gutterLeft: 55,
                gutterTop: 45,
                gutterBottom: 40,
                outofbounds: true,
                linewidth: 3,
                ylabels: false,
                noaxes: true,
                backgroundGrid: false,
                colors: ['red'],
                ymax: 150000,
                scaleInvert: true,
                tickmarks: 'endcircle'
            }
        }).draw();
    };
</script>
